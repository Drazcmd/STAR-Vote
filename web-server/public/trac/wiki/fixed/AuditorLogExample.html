This is a sample log file, containing a log for a simple voting machine protocol with an election coordinator and one voting machine. It is part of a functioning testcase for the auditor. The format is described in AuditorLogFormat.

Contents were copied/pasted from a checkout at ~scrosby/Documents/Research.projects/secure_logging/src/data/votetest.log on July 18 2007. 

% Declare the log entries. <name>/<argcount>
:- declarelogentry(vote/6).
:- declarelogentry(voteauth/5).
:- declarelogentry(electionstart/4).
:- declarelogentry(electionend/4).

%%
%%  Log entries on the election coordinator for start/end of election.
%%
% Columns: 'authenticator', 'host', 'index', 'elecid'

electionstart('aec0', 'ec' , 0 ,'elec0').
  electionend('aec1', 'ec' , 1 ,'elec0').

%%
%% Log entries for the voting machine:
%%
%%     vote(auth,host,index,elecid,voteid,votedata)
%% voteauth(auth,host,index,elecid,voteid)


% Badvote1: Before election initiation.
 voteauth( 'a0','m',0,'elec0','id=0').
     vote( 'a1','m',1,'elec0','id=0',101).

% After election initiation, but vote is not before voteauth.
     vote( 'a2','m',2,'elec0','id=0',102).
 voteauth( 'a3','m',3,'elec0','id=0').
% The only good vote.
 voteauth( 'a4','m',4,'elec0','id=1').
     vote( 'a5','m',5,'elec0','id=1',103).
% Bad votes.
     vote( 'a6','m',6,'elec0','id=1',106).
     vote( 'a7','m',7,'elec0','id=3',104).
 voteauth( 'a8','m',8,'elec0','id=4').
     vote( 'a9','m',9,'elec0','id=4',105).


%%
%% Declare the ordering of the log entries:
%%

%% First, map the hosts names to a number
hostidx('ec',0).
hostidx('m',1).

%% 'ec' means election coordinator  'vm' voting machine.

% raworder(machine1id, machine1index,machine2, machine2minindex)

% Indexes of ec's messages that succeed other ec messages.
raworder(0,0,0,1).
% Indexes of vm's messages that succeed other vm messages.
raworder(1,0,1,1).
raworder(1,1,1,2).
raworder(1,2,1,3).
raworder(1,3,1,4).
raworder(1,4,1,5).
raworder(1,5,1,6).
raworder(1,6,1,7).
raworder(1,7,1,8).
raworder(1,8,1,9).

% Indexes of the vm's messages that succeed the ec's messages.
raworder(0,0,1,2).

% Indexes of the ec's messages that succeed the vm's messages.
raworder(1,0,0,1).
raworder(1,1,0,1).
raworder(1,2,0,1).
raworder(1,3,0,1).
raworder(1,4,0,1).
